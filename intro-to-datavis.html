Getting data ready to plot

annual_esc <- escapement %>%
    separate(sampleDate, c("Year", "Month", "Day"), sep = "-") %>%
    mutate(Year = as.numeric(Year)) %>%
    group_by(Species, SASAP.Region, Year) %>%
    summarize(escapement = sum(DailyCount)) %>%
    filter(Species %in% c("Chinook", "Sockeye", "Chum", "Coho", "Pink"))
## `summarise()` has grouped output by 'Species', 'SASAP.Region'. You can override
## using the `.groups` argument.

Plotting data with ggplot

Essential components

## Options 1
ggplot(data = annual_esc,
       aes(x = Species,
           y = escapement)) +
  geom_col()

## Option 2
ggplot(data = annual_esc)+
       geom_col(aes(x = Species,
                    y = escapement))

## Option 3
ggplot() +
  geom_col(data = annual_esc,
           aes(x = Species,
               y = escapement))

Different geoms

Boxplot

annual_esc %>% 
  filter(Year == 1974,
         Species %in% c("Chum","Pink")) %>% 
  ggplot(aes(x = Species,
             y = escapement)) +
  geom_boxplot(width = 0.4)

Line and Point

annual_esc %>% 
  filter(Species == "Sockeye",
         SASAP.Region == "Bristol Bay") %>% 
  ggplot(aes(x = Year,
             y = escapement)) +
  geom_line() +
  geom_point()

Customizing Plots

fill = blue outside the aes()

ggplot(annual_esc,
       aes(x = Species,
           y = escapement,
           fill = "blue"))+
  geom_col()

Adding color by region

ggplot(annual_esc,
       aes(x = Species,
           y = escapement,
           fill = SASAP.Region))+
  geom_col()
